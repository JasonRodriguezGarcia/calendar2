Asumiendo que vacacionesData es ese array que recibes del backend:

import * as XLSX from "xlsx";
import { saveAs } from "file-saver";

function exportVacacionesToExcel(vacacionesData) {
    // 1. Convertimos JSON ‚Üí Hoja Excel
    const worksheet = XLSX.utils.json_to_sheet(vacacionesData);

    // 2. Creamos el libro de Excel
    const workbook = {
        Sheets: { "Vacaciones": worksheet },
        SheetNames: ["Vacaciones"]
    };

    // 3. Lo convertimos a un buffer
    const excelBuffer = XLSX.write(workbook, {
        bookType: "xlsx",
        type: "array",
    });

    // 4. Lo descargamos
    const blob = new Blob([excelBuffer], {
        type: "application/vnd.openxmlformats-officedocument.spreadsheetml.sheet",
    });

    saveAs(blob, "vacaciones.xlsx");
}
